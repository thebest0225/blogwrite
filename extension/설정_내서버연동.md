# 내 서버 연동 (작업 기록·내 글 보관함 저장)

구글 시트보다 훨씬 간단합니다. **파일 하나 서버에 올리고, 그 주소를 확장에 넣으면 끝.**
확장이 그 주소로 **POST(저장) / GET(조회)** 를 합니다.

---

## 방법 A) PHP (제일 쉬움 — 대부분의 웹호스팅)

### 1. 아래 코드를 `save.php` 로 저장해서 서버에 업로드
(예: `public_html/blog/save.php` → 주소는 `https://내도메인.com/blog/save.php`)

```php
<?php
header('Access-Control-Allow-Origin: *');
header('Access-Control-Allow-Headers: *');
header('Content-Type: application/json; charset=utf-8');

$TOKEN = "";                      // 비워두면 토큰 없이 작동(원하면 아무 문자열)
$FILE  = __DIR__ . '/records.json';

function load($f){ if(!file_exists($f)) return []; $j=json_decode(@file_get_contents($f), true); return is_array($j)?$j:[]; }

if ($_SERVER['REQUEST_METHOD'] === 'POST') {
  $d = json_decode(file_get_contents('php://input'), true);
  if (!is_array($d)) $d = [];
  if ($TOKEN !== '' && (($d['token'] ?? '') !== $TOKEN)) { echo json_encode(['ok'=>false,'error'=>'bad token']); exit; }
  $rec = [
    'date'     => date('c'),
    'type'     => $d['type']     ?? '',
    'title'    => $d['title']    ?? '',
    'url'      => $d['url']      ?? '',
    'keyword'  => $d['keyword']  ?? '',
    'summary'  => $d['summary']  ?? '',
    'platform' => $d['platform'] ?? '',
    'images'   => is_array($d['images'] ?? null) ? implode(' ', $d['images']) : ($d['images'] ?? ''),
    'body'     => $d['body']     ?? ''
  ];
  $fp = fopen($FILE, 'c+'); flock($fp, LOCK_EX);
  $arr = load($FILE); $arr[] = $rec;
  ftruncate($fp, 0); rewind($fp);
  fwrite($fp, json_encode($arr, JSON_UNESCAPED_UNICODE));
  flock($fp, LOCK_UN); fclose($fp);
  echo json_encode(['ok'=>true]); exit;
}

// GET → 목록 반환
if ($TOKEN !== '' && (($_GET['token'] ?? '') !== $TOKEN)) { echo json_encode(['records'=>[]]); exit; }
echo json_encode(['records'=>array_reverse(load($FILE))], JSON_UNESCAPED_UNICODE);
```

### 2. 확장 설정에 입력
- 설정 → **☁️ 저장소 연동 → 저장 API URL** 에 `https://내도메인.com/blog/save.php` 입력
- 비밀 토큰: 위 코드에서 `$TOKEN`을 비워뒀으면 **비워두기**
- 저장 → 끝. (`records.json` 파일이 자동 생성돼 거기에 쌓입니다)

> 기록 전체 보기: 서버의 `records.json` 파일을 열거나, 브라우저로 `save.php` 주소를 열면 JSON으로 보입니다.

## 방법 B) Node.js (Express)

```js
// server.js  →  node server.js  (npm i express)
const express = require("express");
const fs = require("fs");
const path = require("path");
const app = express();
app.use(express.text({ type: "*/*", limit: "5mb" })); // text/plain 수신

const TOKEN = "";
const FILE = path.join(__dirname, "records.json");
const load = () => { try { return JSON.parse(fs.readFileSync(FILE, "utf8")); } catch { return []; } };

app.use((req, res, next) => { res.set("Access-Control-Allow-Origin", "*"); next(); });

app.post("/save", (req, res) => {
  let d = {}; try { d = JSON.parse(req.body || "{}"); } catch {}
  if (TOKEN && d.token !== TOKEN) return res.json({ ok: false, error: "bad token" });
  const arr = load();
  arr.push({
    date: new Date().toISOString(), type: d.type || "", title: d.title || "", url: d.url || "",
    keyword: d.keyword || "", summary: d.summary || "", platform: d.platform || "",
    images: Array.isArray(d.images) ? d.images.join(" ") : (d.images || ""), body: d.body || ""
  });
  fs.writeFileSync(FILE, JSON.stringify(arr));
  res.json({ ok: true });
});

app.get("/save", (req, res) => {
  if (TOKEN && req.query.token !== TOKEN) return res.json({ records: [] });
  res.json({ records: load().reverse() });
});

app.listen(3000, () => console.log("http://localhost:3000/save"));
```
- 확장 설정의 **저장 API URL** 에 `https://내서버.com/save` (또는 `http://내서버IP:3000/save`) 입력

---

## 동작 방식 (참고)
- 확장이 글을 발행/보관할 때 → 그 주소로 **POST**(JSON 한 건 저장)
- 링크 걸 때/보관함 열 때 → **GET**(전체 목록 조회) → 관련 내 글을 먼저 링크

## 보안
- 공개 URL이라 누구나 접근 가능하니, 신경 쓰이면 `$TOKEN`/`TOKEN`에 아무 문자열을 넣고, 확장 설정의 **비밀 토큰**에 같은 값을 넣으세요.
